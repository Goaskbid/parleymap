name: ParleyMap nightly refresh

on:
  workflow_dispatch:
  schedule:
    - cron: "19 3 * * *"

permissions:
  contents: write

concurrency:
  group: parleymap-nightly-refresh
  cancel-in-progress: false

jobs:
  nightly:
    runs-on: ubuntu-latest
    timeout-minutes: 20
    steps:
      - name: Checkout full repository history
        uses: actions/checkout@v4
        with:
          ref: main
          fetch-depth: 0

      - name: Setup Node.js
        uses: actions/setup-node@v4
        with:
          node-version: "20"

      - name: Canonical repair before refresh
        env:
          REQUIRE_ADSENSE_READY: "false"
        run: node scripts/parleymap-canonical-rescue.cjs

      - name: Safe official-source refresh
        run: node nightly-refresh.mjs

      - name: Canonical repair after refresh
        env:
          REQUIRE_ADSENSE_READY: "false"
        run: node scripts/parleymap-canonical-rescue.cjs

      - name: Show summary
        if: always()
        run: |
          if [ -f data/diagnostics/LATEST_RUN_SUMMARY.md ]; then
            cat data/diagnostics/LATEST_RUN_SUMMARY.md >> "$GITHUB_STEP_SUMMARY"
            cat data/diagnostics/LATEST_RUN_SUMMARY.md
          fi

      - name: Upload nightly files and diagnostics
        if: always()
        uses: actions/upload-artifact@v4
        with:
          name: parleymap-nightly-files
          path: |
            index.html
            ads.txt
            data/demo.json
            data/crawler/*
            data/diagnostics/*

      - name: Commit nightly canonical repair
        uses: stefanzweifel/git-auto-commit-action@v5
        with:
          commit_message: "Run ParleyMap nightly canonical repair"
          file_pattern: "index.html ads.txt data/demo.json data/crawler/* data/diagnostics/* privacy.html impressum.html about.html contact.html methodology.html data-sources.html"
